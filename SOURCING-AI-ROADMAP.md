function matchUserProductToShoppingItem(itemName, products) {
  return products
    .filter(p => p.in_stock)
    .map(p => ({
      ...p,
      matchScore: calculateMatchScore(itemName, p)
    }))
    .sort((a, b) => b.matchScore - a.matchScore)
    .slice(0, 3); // Top 3
}
